双链表

实现

class DoubleLink<T>{
    private Node<T> hNode;
    private int mCount;


    class Node<T>{
        private Node prev;
        private Node next;
        private T value;
        public Node(T value,Node prev,Node next){
            this.prev = prev;
            this.next = next;
            this.value = value;
        }
    }
    public DoubleLink(){
        hNode = new Node<T>(null, null, null);       
        mCount = 0;
    }
    public int size(){
        return mCount;
    }
    public boolean isEmpty(){
        return mCount == 0;
    }
    private Node<T> getNode(int index){
        if(index<0||index>=mCount)throw new IndexOutOfBoundsException();//******/
        if(index<=mCount/2){
            Node iNode = hNode.next;
            for(int i=0;i<index;i++){
                iNode = iNode.next;
            }
            return iNode;
        }
        else{
            int rindex = mCount-index-1;
            Node rNode = hNode.prev;
            for(int i=0;i<rindex;i++){
                rNode = rNode.prev;
            }
            return rNode;
        }
    }
    public T get(int index){
        return getNode(index).value;
    }
    public T getFirst(){
        return getNode(0).value;
    }
    public T getLast(){
        return getNode(mCount-1).value;
    }
    //插入到index节点之前
    public void insert(int index,T value){
        if (mCount==0) {
            Node<T> node = new Node<T>(value, hNode, hNode);
            hNode.next = node;
            hNode.prev = node;
            mCount++;
            return ;
        }
        if(index==mCount-1){
            //Node<T> tnode = getNode(index);  这一步可以省略
            Node<T> node = new Node<T>(value,hNode.prev,hNode);
            hNode.prev.next = node;
            hNode.prev = node;
            mCount++;
            return;
        }
        Node<T> node = getNode(index);
        Node<T> tNode = new Node<T>(value,node,node.next);
        node.next.prev = tNode;
        node.next = tNode;
        mCount++;
        return;
    }
    public void insertFirst(T value){
        Node<T> node = new Node<T>(value,hNode,hNode.next);
        hNode.next.prev = node;
        hNode.next = node; 
        mCount++;
    }
    public void appendLast(T value){
        insert(mCount-1, value);
    }
    public void delete(int index){
        Node<T> node = getNode(index);
        node.prev.next = node.next;
        node.next.prev = node.prev;
        node = null;//java的辣鸡回收机制，会回收无法访问的内存
        mCount--;
    }
    public void deleteFirst(){
        delete(0);
    }
    public void deleteLast(){
        delete(mCount-1);
    }
}

import java.lang.reflect.Array;

public class GeneralArrayStack<T> {

    private static final int DEFAULT_SIZE = 12;
    private T[] mArray;
    private int count;

    public GeneralArrayStack(Class<T> type) {
        this(type, DEFAULT_SIZE);
    }//Class<T> T的类型类，不能通过一般的反射手段得到

    public GeneralArrayStack(Class<T> type, int size) {
        // 不能直接使用mArray = new T[DEFAULT_SIZE];
        mArray = (T[]) Array.newInstance(type, size);
        count = 0;
    }

    // 将val添加到栈中
    public void push(T val) {
        mArray[count++] = val;
    }

    // 返回“栈顶元素值”
    public T peek() {
        return mArray[count-1];
    }

    // 返回“栈顶元素值”，并删除“栈顶元素”
    public T pop() {
        T ret = mArray[count-1];
        count--;
        return ret;
    }

    // 返回“栈”的大小
    public int size() {
        return count;
    }

    // 返回“栈”是否为空
    public boolean isEmpty() {
        return size()==0;
    }

    // 打印“栈”
    public void PrintArrayStack() {
        if (isEmpty()) {
            System.out.printf("stack is Empty\n");
        }

        System.out.printf("stack size()=%d\n", size());

        int i=size()-1;
        while (i>=0) {
            System.out.println(mArray[i]);
            i--;
        }
    }
}

树

一些基本定义

  1. 叶子：出度为0的节点
  2. 深度：根节点到当前节点的长度(即线段数)
  3. 高度：最深节点的深度+1

二叉树

定义

所有节点出度小于等于2

性质

  1. 在二叉树的i层上至多有2的i次方个节点

  2. 高度为k(k>=1)的二叉树上至多含有2的k次方-1个节点

  3. 二叉树的高度为
    $$
    [\log_{2}n]+1
    $$
    n为节点数

  4. 任意二叉树 N0 = N2 +1

class Main{

public class Heap<T extends Comparable<T>> {
    private ArrayList<T> heap;

    public Heap() {
        heap = new ArrayList<T>();
    }
    public void printAll(){
        for(int i=0;i<heap.size();i++){
            System.out.println(heap.get(i));
        }
    }
    private void filterup(int start) {
        int c = start;// current
        int p = (c - 1) / 2;
        while (c > 0) {
            if (heap.get(c).compareTo(heap.get(p)) > 0) {
                exchange(c, p);
                c = p;
                p = (p - 1) / 2;
            } else
                break;
        }
    }

    public void insert(T t) {
        int size = heap.size();
        heap.add(t);// 插在末尾
        filterup(size);// 向上调整
    }

    private void filterdown(int start,int end){
        int c = start;
        int l = c*2+1;
        while(l<=end){
            int tem = heap.get(l).compareTo(heap.get(l+1));//这里l+1可能越界
            if(l<end&&tem<0) l++;
            tem = heap.get(c).compareTo(heap.get(l));
            if(tem>0)break;
            else{
                exchange(c, l);
                c = l;
                l = c*2+1;
            }
        }
    }
    public void delete(T t) {
        if (heap.isEmpty() == true)
            return;

        // 获取data在数组中的索引
        int index = heap.indexOf(t);
        if (index == -1)
            return;

        int size = heap.size();
        heap.set(index, heap.get(size - 1));// 用最后元素填补
        heap.remove(size - 1); // 删除最后的元素
        if (heap.size() > 1)filterdown(index, heap.size() - 1); // 从index号位置开始自上向下调整为最小堆
    }

5.哈夫曼树

一些定义

  1. 节点间的路径长度：连接两节点的路径上的分支数
  2. 节点的路径长度：根节点到该节点路径上的分支数
  3. 树加权路径长度：所有叶子节点加权路径长度之和
  4. 哈夫曼树（二叉树）：树加权路径长度最小的那种二叉树

通常用来解决编码问题
